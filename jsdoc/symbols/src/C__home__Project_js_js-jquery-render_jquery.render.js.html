  7 
  8 $.fn.extend({
  9 	/**
 10 	 * Function to render HTML from template and data
 11 	 * @param {object} data Collection of data ( array or object )
 12 	 * @return {object} jQuery object which contains rendered HTML.
 13 	 * @example $("script#template-example").render( data ).appendTo( $("#Container") );
 14 	 */
 15 	render:function( data ){
 16 		var tmpl = this.html(), result = "",
 17 			create = function(t,d){
 18 				var m = t.match(/{{(.+?)}}/g) || [], r = t;
 19 				$.each( m, function(i,k){
 20 					var v = d[ k.replace(/{|}/g, "") ];
 21 					r = r.replace( k, ( v === undefined ) ? "" : v );
 22 				});
 23 				return r;
 24 			};
 25 		if( data === undefined ){ return $( tmpl.replace(/{{(.+?)}}/g, "") ); }
 26 		if( !(data instanceof Array) ){ return $( create(tmpl, data) ); }
 27 		$.each( data, function(i,o){ result += create(tmpl,o); } );
 28 		return $(result);
 29 	}
 30 });
